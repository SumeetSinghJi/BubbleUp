\DoxyCodeLine{\#\#\ DESCRIPTION}
\DoxyCodeLine{}
\DoxyCodeLine{BubbleUP\ is\ a\ combined\ C++\ 2D\ Game\ Engine\ with\ a\ built\ in\ feature-\/rich\ demonstration\ 2D\ game\ built\ }
\DoxyCodeLine{ontop\ of\ the\ SDL\ Framework.\ It\ includes\ classes\ for\ constructing\ interactive\ Buttons\ and\ fields\ to\ }
\DoxyCodeLine{create\ stunning\ GUI,\ Forms\ and\ HUD.\ Includes\ countless\ game\ entities\ and\ level\ editor\ to\ create\ unique\ }
\DoxyCodeLine{stunning\ 2D\ games.\ Has\ built\ in\ scenes\ for\ achievements,\ leaderboards,\ multiplayer,\ and\ more.\ You\ can\ }
\DoxyCodeLine{download\ and\ easily\ modify\ this\ open-\/source\ code\ to\ create\ your\ own\ game\ by\ following\ the\ }
\DoxyCodeLine{instructions\ here.}
\DoxyCodeLine{}
\DoxyCodeLine{}
\DoxyCodeLine{\#\#\#\ GUI}
\DoxyCodeLine{}
\DoxyCodeLine{Uses\ a\ BaseButton\ class\ to\ encapsulate\ logic\ for\ different\ GUI\ button\ objects\ derived\ from\ SDL\ }
\DoxyCodeLine{Framework\ for\ various\ GUI\ elements,\ e.g.,\ volume\ sliders,\ input\ fields,\ clickable\ radios,\ checkboxes,}
\DoxyCodeLine{\ and\ buttons.\ These\ are\ initialized\ within\ a\ vector\ to\ hold\ all\ the\ different\ buttons,\ enabling\ }
\DoxyCodeLine{\ navigation\ through\ them\ with\ keyboard\ movement,\ highlighting,\ and\ building\ a\ GUI\ with\ buttons,\ }
\DoxyCodeLine{\ rendering\ text/draws,\ and\ handles.}

\DoxyCodeLine{}
\DoxyCodeLine{\#\#\#\ GAME}

\DoxyCodeLine{Obstacles\ (boundaries).\ Each\ has\ its\ own\ logic\ for\ movement\ and\ collision.\ A\ basic\ displacement\ }
\DoxyCodeLine{camera\ is\ used\ to\ draw\ the\ location\ of\ entities.\ The\ existing\ game,\ "{}Bubble\ UP,"{}\ is\ a\ simple\ }
\DoxyCodeLine{"{}reach\ the\ highest\ score"{}\ game\ by\ collecting\ valuable\ items\ while\ trying\ to\ avoid\ touching\ }
\DoxyCodeLine{enemies,\ all\ while\ navigating\ obstacles\ blocking\ the\ way.}

\DoxyCodeLine{}
\DoxyCodeLine{\#\#\#\ LEADERBOARDS}

\DoxyCodeLine{any\ platform.\ Achievements\ can\ be\ shared\ online\ with\ other\ players\ and\ leaderboard\ statistics\ }
