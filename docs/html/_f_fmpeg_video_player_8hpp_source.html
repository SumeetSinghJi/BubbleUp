167 FFmpegVideoPlayer(const char *videoFile, const char *musicFile, SDL_Window *window, SDL_Renderer *renderer)
168 : videoFile(videoFile), musicFile(musicFile), window(window), renderer(renderer), formatCtx(nullptr),
169 videoStreamIndex(-1), audioStreamIndex(-1), videoCodec(nullptr), audioCodec(nullptr),
170 videoCodecCtx(nullptr), audioCodecCtx(nullptr), music(nullptr)
171 {
172 avformat_network_init();
173
174 // Open the input file
175 if (avformat_open_input(&formatCtx, videoFile, nullptr, nullptr) < 0)
176 {
177 char errbuf[AV_ERROR_MAX_STRING_SIZE];
178 av_strerror(errno, errbuf, AV_ERROR_MAX_STRING_SIZE);
179 std::cerr << "Error: Could not open Video file: " << videoFile << " - " << errbuf << std::endl;
180 return;
181 }
182
183 // // Debug: Printing stream information for debugging
184 // std::cout << "Debug: Printing stream information for debugging:" << std::endl;
185 // av_dump_format(formatCtx, 0, videoFile, 0);
186
187 // Find the stream information
188 if (avformat_find_stream_info(formatCtx, nullptr) < 0)
189 {
190 char errbuf[AV_ERROR_MAX_STRING_SIZE];
191 av_strerror(errno, errbuf, AV_ERROR_MAX_STRING_SIZE);
192 std::cerr << "Error: Could not find stream information: " << errbuf << std::endl;
193 avformat_close_input(&formatCtx); // Close input on failure
194 return;
195 }
196
197 // Find the first video stream and audio stream
198 for (unsigned int i = 0; i < formatCtx->nb_streams; i++)
199 {
200 if (formatCtx->streams[i]->codecpar->codec_type == AVMEDIA_TYPE_VIDEO && videoStreamIndex < 0)
201 {
202 videoStreamIndex = i;
203 std::cout << "Found Video Stream: Index " << videoStreamIndex << std::endl;
204 // // Debug: Print video codec information
205 // std::cout << "Debug: Video Codec Information:" << std::endl;
206 // av_dump_format(formatCtx, videoStreamIndex, videoFile, 0);
207 }
208 else if (formatCtx->streams[i]->codecpar->codec_type == AVMEDIA_TYPE_AUDIO && audioStreamIndex < 0)
209 {
210 audioStreamIndex = i;
211 std::cout << "Found Audio Stream: Index " << audioStreamIndex << std::endl;
212 // // Debug: Print audio codec information
213 // std::cout << "Debug: Audio Codec Information:" << std::endl;
214 // av_dump_format(formatCtx, audioStreamIndex, videoFile, 0); // Changed this line
215 }
216 }
217
218 if (videoStreamIndex == -1 && audioStreamIndex == -1)
219 {
220 std::cerr << "Error: Could not find video or audio stream." << std::endl;
221 avformat_close_input(&formatCtx); // Close input on failure
222 return;
223 }
224
225 // Open video codec
226 if (videoStreamIndex >= 0)
227 {
228 videoCodecCtx = avcodec_alloc_context3(nullptr);
229 if (!videoCodecCtx)
230 {
231 std::cerr << "Error: Could not allocate video codec context." << std::endl;
232 avformat_close_input(&formatCtx); // Close input on failure
233 return;
234 }
235 avcodec_parameters_to_context(videoCodecCtx, formatCtx->streams[videoStreamIndex]->codecpar);
236 videoCodec = avcodec_find_decoder(videoCodecCtx->codec_id);
237 if (!videoCodec)
238 {
239 std::cerr << "Error: Could not find video codec." << std::endl;
240 avformat_close_input(&formatCtx); // Close input on failure
241 return;
242 }
243 if (avcodec_open2(videoCodecCtx, videoCodec, nullptr) < 0)
244 {
245 std::cerr << "Error: Could not open video codec." << std::endl;
246 avformat_close_input(&formatCtx); // Close input on failure
247 return;
248 }
249 }
250
251 // Open audio codec
252 if (audioStreamIndex >= 0)
253 {
254 audioCodecCtx = avcodec_alloc_context3(nullptr);
255 if (!audioCodecCtx)
256 {
257 std::cerr << "Error: Could not allocate audio codec context." << std::endl;
258 avformat_close_input(&formatCtx); // Close input on failure
259 return;
260 }
261 avcodec_parameters_to_context(audioCodecCtx, formatCtx->streams[audioStreamIndex]->codecpar);
262 audioCodec = avcodec_find_decoder(audioCodecCtx->codec_id);
263 if (!audioCodec)
264 {
265 std::cerr << "Error: Could not find audio codec." << std::endl;
266 avformat_close_input(&formatCtx); // Close input on failure
267 return;
268 }
269 if (avcodec_open2(audioCodecCtx, audioCodec, nullptr) < 0)
270 {
271 std::cerr << "Error: Could not open audio codec." << std::endl;
272 avformat_close_input(&formatCtx); // Close input on failure
273 return;
274 }
275
276 // Initialize SDL Mixer
277 if (Mix_OpenAudio(44100, MIX_DEFAULT_FORMAT, 2, 4096) == -1)
278 {
279 std::cerr << "Error: Failed to open audio channel: " << Mix_GetError() << std::endl;
280 return;
281 }
282
283 // Initialize music with SDL Mixer
284 music = Mix_LoadMUS(musicFile);
285 if (!music)
286 {
287 std::cerr << "Error: Failed to load music file: " << Mix_GetError() << std::endl;
288 return;
289 }
290 }

345void playVideo()
346{
347 // Check if constructor initialised successfully
348 if (!formatCtx || !videoCodecCtx || !audioCodecCtx)
349 {
350 std::cerr << "Error: FFmpeg initialization failed." << std::endl;
351 return;
352 }
353
354 // Create SDL texture
355 SDL_Texture *texture = SDL_CreateTexture(renderer, SDL_PIXELFORMAT_YV12, SDL_TEXTUREACCESS_STREAMING,
356 videoCodecCtx->width, videoCodecCtx->height);
357 if (texture == nullptr)
358 {
359 std::cerr << "Error: Could not create SDL texture: " << SDL_GetError() << std::endl;
360 return;
361 }
362
363 // Allocate frame
364 AVFrame *frame = av_frame_alloc();
365 AVPacket packet;
366
367 // Calculate delay to achieve desired frame rate
368 double frame_delay = 1.0 / av_q2d(formatCtx->streams[videoStreamIndex]->avg_frame_rate);
369
370 // Start audio playback
371 playAudio();
372
373 // Start video playback loop
374 bool end_of_stream = false; // Flag to track end of video stream
375 while (!end_of_stream) // Loop until end of stream
376 {
377 // Read video frames
378 while (!end_of_stream && av_read_frame(formatCtx, &packet) >= 0)
379 {
380 if (packet.stream_index == videoStreamIndex)
381 {
382 avcodec_send_packet(videoCodecCtx, &packet);
383 avcodec_receive_frame(videoCodecCtx, frame);
384
385 SDL_RenderClear(renderer);
386
387 // Update the video texture
388 SDL_UpdateYUVTexture(texture, nullptr,
389 frame->data[0], frame->linesize[0],
390 frame->data[1], frame->linesize[1],
391 frame->data[2], frame->linesize[2]);
392
393 // Render the video frame
394 SDL_RenderCopy(renderer, texture, nullptr, nullptr);
395 SDL_RenderPresent(renderer);
396
397 // Delay the video so that it plays at the same speed as video stream
398 SDL_Delay((Uint32)(frame_delay * 1000)); // Convert to milliseconds
399
400 /*
401 It's necessary to have an event loop here.
402 The reason is while the video is playing, if you move the window
403 SDL expects their to be a key input handle loop logic to take care of this event
404 but because their are no case for events it will crash.
405
406 Thus we initialise a fake event loop and also pressing any button skips the cutscene which
407 can be customised
408 */
409 SDL_Event event;
410 while (SDL_PollEvent(&event))
411 {
412 switch (event.type)
413 {
414 case SDL_QUIT:
415 case SDL_KEYDOWN:
416 case SDL_CONTROLLERBUTTONDOWN:
417 case SDL_MOUSEBUTTONDOWN:
418 case SDL_FINGERDOWN:
419 end_of_stream = true;
420 break;
421 }
422 }
423 }
424 av_packet_unref(&packet);
425 }
426
427 // When video stream ends, quit and return
428 if (av_read_frame(formatCtx, &packet) == AVERROR_EOF)
429 end_of_stream = true;
430 }
431
432 // Cleanup
433 av_frame_free(&frame);
434 SDL_DestroyTexture(texture);
435}
