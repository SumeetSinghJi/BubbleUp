731 void render_button_text(const std::string &text, int x, int y, Uint8 redText, Uint8 greenText, Uint8 blueText, Uint8 alphaText)
732 {
733 // alpha value 1 = transparent, alpha value 0 breaks and becomes transparent again. Workaround is to just use alpha 1;
734 if (buttonFont)
735 {
736 SDL_Color textColor = {redText, greenText, blueText, alphaText};
737 SDL_Surface *textSurface = TTF_RenderUTF8_Blended(buttonFont, text.c_str(), textColor);
738
739 if (textSurface)
740 {
741 int textWidth, textHeight;
742 // Calculate textWidth and textHeight using TTF_SizeText
743 TTF_SizeText(buttonFont, text.c_str(), &textWidth, &textHeight);
744
745 SDL_Texture *textTexture = SDL_CreateTextureFromSurface(renderer, textSurface);
746 SDL_FreeSurface(textSurface);
747
748 if (textTexture)
749 {
750 SDL_Rect textRect = {x, y, textWidth, textHeight};
751 SDL_RenderCopy(renderer, textTexture, nullptr, &textRect);
752 }
753 else
754 {
755 std::cout << "Error: Unable to render text to surface" << TTF_GetError() << std::endl;
756 }
757 SDL_DestroyTexture(textTexture);
758 }
759 }
760 else
761 {
762 std::cout << "Error: Button: " << buttonLabel << " Unable to render_button_text: " << TTF_GetError() << std::endl;
763 }
764 }

773 virtual void render_button_rect()
774 {
775 // If no Texture, draw boxes
776 if (buttonTexturePath.empty())
777 {
778 if (isSelected)
779 {
780 /*
781 Borders will appear as Grey/Yellow box if texture/Font is same color (Grey/Yellow) this is
782 because the order of events are
783
784 1. draw large Grey/Yellow (color depending on selected) box with pixels large e.g. 100
785 2. draw button with pixels smaller e.g. 98
786 3. Thus the large Grey/Yellow box has it's insides replaced with the texture/Button Text
787
788 thus this creates a border effect)
789 */
790 // Draw - YELLOW border
791 SDL_SetRenderDrawColor(renderer, 255, 255, 0, 255); // yellow
792 SDL_Rect borderRect = {buttonRect.x - 2, buttonRect.y - 2, buttonRect.w + 4, buttonRect.h + 4};
793 SDL_RenderFillRect(renderer, &borderRect);
794 }
795 else
796 {
797 // Draw - GREY border
798 SDL_SetRenderDrawColor(renderer, 255, 255, 255, 255); // Grey
799 SDL_Rect borderRect = {buttonRect.x - 2, buttonRect.y - 2, buttonRect.w + 4, buttonRect.h + 4};
800 SDL_RenderFillRect(renderer, &borderRect);
801 }
802 // Draw button
803 SDL_SetRenderDrawColor(renderer, buttonColor.r, buttonColor.g, buttonColor.b, buttonColor.a);
804 SDL_RenderFillRect(renderer, &buttonRect);
805 }
806 else
807 {
808 if (isSelected)
809 {
810 /*
811 Borders will appear as Grey/Yellow box if texture/Font is same color (Grey/Yellow) this is
812 because the order of events are
813
814 1. draw large Grey/Yellow (color depending on selected) box with pixels large e.g. 100
815 2. draw button with pixels smaller e.g. 98
816 3. Thus the large Grey/Yellow box has it's insides replaced with the texture/Button Text
817
818 thus this creates a border effect)
819 */
820 // Draw - YELLOW border
821 SDL_SetRenderDrawColor(renderer, 255, 255, 0, 255); // yellow
822 SDL_Rect borderRect = {buttonRect.x - 2, buttonRect.y - 2, buttonRect.w + 4, buttonRect.h + 4};
823 SDL_RenderFillRect(renderer, &borderRect);
824 }
825 // Draw Texture
826 SDL_RenderCopy(renderer, buttonTexture, nullptr, &buttonRect);
827 }
828 // Draw text
829 render_button_text(buttonLabel, buttonRect.x + 30, buttonRect.y + 15, buttonTextColor.r, buttonTextColor.g, buttonTextColor.b, buttonTextColor.a);
830 }

877 static double find_euclidean_distance(int x1, int y1, int x2, int y2)
878 {
879 double results = std::sqrt((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));
880 return results;
881 }

896 static void find_nearest_button(std::vector<BaseButton *> sceneVector, const std::string buttonDirection)
897 {
898 float minDistance = std::numeric_limits<float>::max();
899 int currentButtonXpos = 0;
900 int currentButtonYpos = 0;
901
902 BaseButton *nearestButton = nullptr;
903
904 // Get the position of the selected button
905 for (BaseButton *button : sceneVector)
906 {
907 if (button->is_selected())
908 {
909 currentButtonXpos = button->get_rect().x + button->get_rect().w / 2;
910 currentButtonYpos = button->get_rect().y + button->get_rect().h / 2;
911 break;
912 }
913 }
914
915 // Prioritize moving in the specified direction
916 for (BaseButton *button : sceneVector)
917 {
918 if (!button->is_selected())
919 {
920 if (buttonDirection == "UP" && button->get_rect().y < currentButtonYpos)
921 {
922 float dist = find_euclidean_distance(currentButtonXpos, currentButtonYpos,
923 button->get_rect().x + button->get_rect().w / 2,
924 button->get_rect().y + button->get_rect().h / 2);
925
926 if (dist < minDistance)
927 {
928 minDistance = dist;
929 nearestButton = button;
930 }
931 }
932 else if (buttonDirection == "DOWN" && button->get_rect().y > currentButtonYpos)
933 {
934 float dist = find_euclidean_distance(currentButtonXpos, currentButtonYpos,
935 button->get_rect().x + button->get_rect().w / 2,
936 button->get_rect().y + button->get_rect().h / 2);
937
938 if (dist < minDistance)
939 {
940 minDistance = dist;
941 nearestButton = button;
942 }
943 }
944 else if (buttonDirection == "LEFT" && button->get_rect().x < currentButtonXpos)
945 {
946 float dist = find_euclidean_distance(currentButtonXpos, currentButtonYpos,
947 button->get_rect().x + button->get_rect().w / 2,
948 button->get_rect().y + button->get_rect().h / 2);
949
950 if (dist < minDistance)
951 {
952 minDistance = dist;
953 nearestButton = button;
954 }
955 }
956 else if (buttonDirection == "RIGHT" && button->get_rect().x > currentButtonXpos)
957 {
958 float dist = find_euclidean_distance(currentButtonXpos, currentButtonYpos,
959 button->get_rect().x + button->get_rect().w / 2,
960 button->get_rect().y + button->get_rect().h / 2);
961
962 if (dist < minDistance)
963 {
964 minDistance = dist;
965 nearestButton = button;
966 }
967 }
968 }
969 }
970
971 // Select the nearest button
972 if (nearestButton != nullptr)
973 {
974 nearestButton->set_selected();
975 }
976 }
