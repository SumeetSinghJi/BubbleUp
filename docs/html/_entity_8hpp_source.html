121 void update_animation()
122 {
123 auto currentTime = std::chrono::steady_clock::now();
124 auto elapsedTime = std::chrono::duration_cast<std::chrono::milliseconds>(currentTime - lastFrameChange).count();
125
126 if (elapsedTime >= animationDelay)
127 {
128 lastFrameChange = currentTime;
129 currentAnimationFrame = (currentAnimationFrame + 1) % walkingTextures.size();
130
131 texturePath = walkingTextures[currentAnimationFrame];
132 }

641 void update_deceleration(float deceleration)
642 {
643 // std::cout << "Decelerating Y Velocity: " << player1.get_xVelocity() << std::endl;
644 // std::cout << "Decelerating X Velocity: " << yVelocity << std::endl;
645
646 if (yVelocity < 0.0f) // move up y--
647 {
648 yVelocity += deceleration; // Increase y-velocity
649 if (yVelocity > 0.0f) // if y++ too much reset to 0.0
650 {
651 yVelocity = 0.0f;
652 }
653 }
654 else if (yVelocity > 0.0f) // moving down y++
655 {
656 yVelocity -= deceleration; // Decrease y-velocity
657 if (yVelocity < 0.0f) // if y-- too much, reset to 0.0
658 {
659 yVelocity = 0.0f;
660 }
661 }
662
663 if (xVelocity < 0.0f) // move left x--
664 {
665 xVelocity += deceleration; // Increase x-velocity
666 if (xVelocity > 0.0f) // if x++ too much, reset to 0.0
667 {
668 xVelocity = 0.0f;
669 }
670 }
671 else if (xVelocity > 0.0f) // move right x++ accelerate
672 {
673 xVelocity -= deceleration; // Decrease x-velocity
674 if (xVelocity < 0.0f)
675 {
676 xVelocity = 0.0f; // if x-- too much, reset to 0.0
677 }
678 }
