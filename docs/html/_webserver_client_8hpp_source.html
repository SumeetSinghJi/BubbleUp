159 void login_account(std::string username, std::string password)
160 {
161 try
162 {
163 // Form data
164 std::string queryURL = "agnisamooh.com";
165 std::string queryProtocol = "http";
166
167 // Form request payload
168 std::string POSTData = "login-username=" + username +
169 "&login-password=" + password;
170
171 // Create request
172 std::string request = "POST /website/signup.html HTTP/1.1\r\n"
173 "Host: agnisamooh.com\r\n"
174 "Content-Type: application/x-www-form-urlencoded\r\n"
175 "Content-Length: " +
176 std::to_string(POSTData.length()) + "\r\n\r\n" +
177 POSTData;
178
179 // Connect to endpoint
180 boost::asio::ip::tcp::resolver::results_type endpoints = resolver.resolve(queryURL, queryProtocol);
181 boost::asio::connect(socket, endpoints);
182
183 // Send the request
184 boost::asio::write(socket, boost::asio::buffer(request.data(), request.size()));
185
186 // Read the response
187 boost::asio::streambuf response;
188 boost::asio::read_until(socket, response, "\r\n");
189
190 // Output the response
191 std::istream responseStream(&response);
192 std::string httpVersion{};
193 responseStream >> httpVersion;
194 int statusCode{};
195 responseStream >> statusCode;
196 std::string statusMessage{};
197 std::getline(responseStream, statusMessage);
198 if (!responseStream || httpVersion.substr(0, 5) != "HTTP/")
199 {
200 std::cerr << "Error: Invalid server response" << std::endl;
201 return;
202 }
203 if (statusCode != 200)
204 {
205 std::cerr << "Error: Incorrect response code " << statusCode << std::endl;
206 return;
207 }
208
209 // Search for the word "pass" or "fail" in the response line by line
210 std::string line;
211 bool success = false;
212 while (std::getline(responseStream, line))
213 {
214 std::cout << line << std::endl; // Print the response line
215 if (line.find("pass") != std::string::npos)
216 {
217 successResponse = true;
218 break;
219 }
220 else if (line.find("fail") != std::string::npos)
221 {
222 successResponse = false;
223 break;
224 }
225 }
226 socket.close();
227 }
228 catch (std::exception &e)
229 {
230 std::cerr << "Error: Exception: " << e.what() << std::endl;
231 }
232 }
